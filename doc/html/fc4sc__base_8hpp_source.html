46 /*
47  * Template meta-programming tool used for checking that a parameter
48  * pack doesn't contain any argument which is convertible to a specified type.
49  * This is used in the bin & coverpoint constructors together with a
50  * static_assert in order to make sure that the user does not provide
51  * multiple arguments which must be provided only once.
52  * For bin -> the name
53  * For coverpoint -> pointer to the parent covergroup
54  */
55 template<typename forbidden, typename head, typename...tail> struct forbid_type {
56  static bool constexpr value = forbid_type<forbidden, tail...>::value &&
57  (!std::is_convertible<head, forbidden>::value);
58 };
59 template<typename forbidden, typename first> struct forbid_type<forbidden, first> :
60  public std::__not_<std::is_convertible<first, forbidden>> {};

91 /*
92  * Macro used to define and automatically register a coverpoint inside a covergroup.
93  * The macro accepts 3 or 4 arguments in the following order:
94  * 1) Type: used for the values of the bins declared in the coverpoint
95  * 2) Name: the name of the coverpoint
96  * 3) Sample Expression: the expression that will be used for sampling this coverpoint
97  * whenever the covergroup is sampled (the type has to be the same as Type argument)
98  * 4) Sample Condition: a condition that has to be true in order for this coverpoint
99  * to be sampled when the covergroup is sampled (the type has to be bool)
100  */
101 #define COVERPOINT(...) GET_CVP_MACRO(__VA_ARGS__, CVP_4, CVP_3)(__VA_ARGS__)
