303 template <class T>
304 class bin_array final : public bin<T>
305 {
306  static_assert(std::is_arithmetic<T>::value, "Type must be numeric!");
307 protected:
308  /* Virtual function used to register this bin inside a coverpoint */
309  virtual void add_to_cvp(coverpoint<T> &cvp) const override
310  {
311  if (this->sparse) {
312  // bin array was defined by using a vector of intervals or values
313  // create a new bin for each value/interval and add it to the coverpoint
314  std::stringstream ss;
315 
316  for (size_t i = 0; i < this->intervals.size(); ++i) {
317  ss << this->name << "[" << i << "]";
318  cvp.bins.push_back(bin<T>(ss.str(), this->intervals[i]));
319  ss.str(std::string()); // clear the stringstream
320  }
321  }
322  else {
323  // bin array was defined by using an interval which needs to be split into
324  // multiple pieces. The interval is found in the this->intervals[0]
325 
326  // FIXME: interval_length is not properly calculated for floating point types
327  T interval_length = (this->intervals[0].second - this->intervals[0].first) + 1;
328 
329  // The if following condition can trigger comparison warnings.
330  // Casting interval_length is not a viable option because if T is float
331  // or double, we will cast away the floating point and lose information!
332  // Nor is casting the count variable to T either, because we might undercast
333  // TODO: find a way to work around this issue
334  // NOTE: A potential fix would be implementing a template specialization
335  // of the bin_array class for floating point types (float/double).
336  if (this->count > interval_length) {
337  // This bin array interval cannot be split into pieces. Add a single
338  // bin containing the whole interval to the coverpoint. We can simply
339  // use this object since it already matches the bin that we need!
340  cvp.bins.push_back(*this);
341  }
342  else {
343  std::stringstream ss;
344  // This bin array interval must be split into pieces.
345  T start = this->intervals[0].first;
346  T stop = this->intervals[0].second;
347  T interval_len = (interval_length + 1) / this->count;
348 
349  for (size_t i = 0; i < this->count; ++i) {
350  ss << this->name << "[" << i << "]";
351  // the last interval, will contain all the extra elements
352  T end = (i == (this->count - 1)) ? stop : start + (interval_len - 1);
353  cvp.bins.push_back(bin<T>(ss.str(), interval(start, end)));
354  start = start + interval_len;
355  ss.str(std::string()); // clear the stringstream
356  }
357  }
358  }
359  }

404 /*
405  * Bin wrapper class used when constructing coverpoint via the COVERPOINT macro.
406  * Under the hood, the macro instantiates a coverpoint using std::initializer_list
407  * as argument. Because the std::initializer_list is limited to one type only,
408  * we cannot directly pass any type of bin we want to the coverpoint. In order to
409  * do that, we use this class which offers implicit cast from any type of bin and
410  * stores it internally as a dynamically allocated object.
411  */
412 template <class T>
413 class bin_wrapper final {
414 private:
415  friend class coverpoint<T>;
416 
417  std::unique_ptr<bin<T>> bin_h;
418  bin<T> *get_bin() const { return bin_h.get(); }
419 
420 public:
421  bin_wrapper(bin_wrapper && r) { bin_h = std::move(r.bin_h); }
422  bin_wrapper() = delete;
423  bin_wrapper(bin_wrapper &) = delete;
424  bin_wrapper& operator=(bin_wrapper &) = delete;
425  bin_wrapper& operator=(bin_wrapper &&) = delete;
426 public:
427  ~bin_wrapper() = default;
428  // Implicit cast to other bin types.
429  bin_wrapper(bin<T> && r) noexcept : bin_h(new bin<T> (std::move(r))) {}
430  bin_wrapper(bin_array<T> && r) noexcept : bin_h(new bin_array<T> (std::move(r))) {}
431  bin_wrapper(illegal_bin<T>&& r) noexcept : bin_h(new illegal_bin<T>(std::move(r))) {}
432  bin_wrapper(ignore_bin<T> && r) noexcept : bin_h(new ignore_bin<T> (std::move(r))) {}
433 };
