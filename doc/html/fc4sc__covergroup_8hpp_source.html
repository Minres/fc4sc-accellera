52  /*
53  * This function registers a coverpoint instance inside this covergroup.
54  * It receives as arguments a pointer to the coverpoint to be registered,
55  * the coverpoint name, the sample expression lambda function and string,
56  * and finally, the sample condition lambda function and string.
57  * Finally, it returns a coverpoint constructed with the given arguments.
58  * The purpose of this function is to be used for coverpoint instantiation
59  * via the COVERPOINT macro and should not be explicitly used!
60  */
61  template<typename T>
62  coverpoint <T> register_cvp(coverpoint <T>* cvp, std::string&& cvp_name,
63  std::function<T()>&& sample_expr, std::string&& sample_expr_str,
64  std::function<bool()>&& sample_cond, std::string&& sample_cond_str) {
65 
66  // NOTE: VERY important! Do not attempt to dereference the cvp pointer in
67  // any way because it points to uninitialized memory!
68  cvg_base::register_cvp(cvp);
69  coverpoint<T> cvp_structure;
70  cvp_structure.has_sample_expression = true;
71  cvp_structure.sample_expression = sample_expr;
72  cvp_structure.sample_condition = sample_cond;
73  cvp_structure.sample_expression_str = sample_expr_str;
74  cvp_structure.sample_condition_str = sample_cond_str;
75  cvp_structure.name = cvp_name;

140  double get_inst_coverage() const
141  {
142  double res = 0;
143  double weights = 0;
144 
145  for (auto &cvp : cvps) {
146  res += cvp->get_inst_coverage() * cvp->option.weight;
147  weights += cvp->option.weight;
148  }
149 
150  if (weights == 0 || cvps.size() == 0 || res == 0)
151  return (this->option.weight == 0) ? 100 : 0;
152 
153  double real = res / weights;
154  return (real >= this->option.goal) ? 100 : real;
155  }

163  double get_inst_coverage(int &bins_covered, int &total) const
164  {
165  double res = 0;
166  double weights = 0;
167  int bins_aux = 0;
168  int total_aux = 0;
169 
170  bins_covered = 0;
171  total = 0;
172 
173  for (auto &cvp : cvps)
174  {
175  res += cvp->get_inst_coverage(bins_aux, total_aux) * cvp->option.weight;
176  bins_covered += bins_aux;
177  total += total_aux;
178  weights += cvp->option.weight;
179  }
180 
181  if (weights == 0 || cvps.size() == 0 || res == 0)
182  {
183  total = 0;
184  bins_covered = 0;
185  return (this->option.weight == 0) ? 100 : 0;
186  }
187 
188  double real = res / weights;
189  return (real >= this->option.goal) ? 100 : real;
190  }
