94  bool check(std::vector<std::vector<size_t>> &found, Head h, Tail... t)
95  {
96 
97  // See if the current value is in its coverpoint, and where
98  std::vector<size_t> bin_indexes = (static_cast<coverpoint<Head> *>(cvps_vec[k - 1]))->get_bin_index(h);
99 
100  // If atleast one bin has it

102 
103  if (!found_in_cvp)
104  {
105  return false;
106  }
107 
108  // Store bins for this coverpoint
109  found.push_back(bin_indexes);
